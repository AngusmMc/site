                # our array that we want to search through
                my_list = [1,3,5,7,9]
                
                def binary_search(arr, item):

                    low = 0                     # low and high keep track of which part of the list we want to search
                    high = len(arr)-1           # the last element
                    
                    while low <= high:          #  while we haven't narrowed it down to one element
                        mid = (low + high) // 2     # check the middle element
                        guess = arr[mid]            # change guess to that middle element

                        if guess == item:               # check if guess is the item we are searching for
                            return mid                  # return the item, we found it!

                        elif guess > item:              # the guess was too high
                            high = mid - 1              # 

                        else:                           # the guess was too low
                            low = mid + 1
                    return None 
                
                print(binary_search(my_list, 3)) # => 1      3 is at index 1
                print(binary_search(my_list, -1)) # => None  the item was not found
